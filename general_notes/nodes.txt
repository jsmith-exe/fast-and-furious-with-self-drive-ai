roscore     #Start the robot master node



Chassis Node:
	
	Jetson:
		roslaunch jetracer jetracer.launch   #Start the robot chassis node
		rostopic echo /cmd_vel    #Display /cmd_vel receives the topic

	Virtual Machine:
		roslaunch jetracer keyboard.launch	#Control the car using the keyboard
		roslaunch jetracer joy.launch	#Control the car using the controller
		sudo nano catkin_ws/src/jetracer_ros/scripts/teleop_joy.py   #Open the remote control program file



Linear Velocity Calibration:

	Jetson:
		roslaunch jetracer calibrate_linear.launch	#Start the linear calibration node

	Virtual Machine:
		rosrun rqt_reconfigure rqt_reconfigure		#Calibrate the car



Camera Node:

	Jetson:
		roslaunch jetracer csi_camera.launch	#Start the camera node
		rosrun web_video_server web_video_server	#Start the web server

	Virtual Machine:
		rosrun rqt_image_view rqt_image_view	#View the camera stream inside the Virtual Machine
		http://localhost:8080/stream?topic=/csi_cam_0/image_raw	#View the stream on a web browser



Lidar Node:

	Jetson:
		roslaunch jetracer lidar.launch 	#Start the Lidar node
		roslaunch jetracer laser_filter.launch        #Start the radar filter node
		roslaunch jetracer slam.launch   #The mapping algorithm uses gmapping by default

	Virtual Machine:
		roslaunch jetracer view_slam.launch     #Display the slam map
